保留两位小数

package main

import (
	"fmt"
	"math"
)

func RoundFloat(val float64, precision int) float64 {
	ratio := math.Pow(10, float64(precision))
	return math.Round(val*ratio) / ratio
}

func main() {
	fmt.Println(RoundFloat(3.1415926, 2))
	fmt.Println(RoundFloat(1024.2325, 1))
}

运行代码输出如下

$  go run main.go

3.14
1024.2
